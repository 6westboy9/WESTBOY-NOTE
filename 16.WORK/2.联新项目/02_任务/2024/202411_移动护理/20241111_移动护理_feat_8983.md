```sql
USE windranger_mnis;
CREATE TABLE IF NOT EXISTS `doc_data_filing_record_collection` (
  `seq_id` bigint(20) unsigned NOT NULL AUTO_INCREMENT,
  `inhos_code` varchar(20) NOT NULL COMMENT '住院号',
  `document_code` varchar(20) NOT NULL COMMENT '文书号',
  `status` int(11) default '0' COMMENT '状态 0.未采集 1.已采集',
  `create_person` varchar(20) DEFAULT NULL COMMENT '创建人',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  `update_person` varchar(20) DEFAULT NULL COMMENT '更新人',
  `update_time` datetime DEFAULT NULL COMMENT '更新时间',
  PRIMARY KEY (`seq_id`),
  KEY `idx_inhos_code-document_code` (`inhos_code`,`document_code`),
  KEY `idx_document_code` (`document_code`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8
    comment '无纸化归档文书采集状态表';
```
