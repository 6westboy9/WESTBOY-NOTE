use mnis
let rootConfig = db.mnisReportConfig.findOne({})

function handle(rate) {
    if (Array.isArray(rate) && rate.length > 0) {
        let isSet = true;
        for (let i = 0; i < rate.length; i++) {
            if (rate[i].valid) {
                isSet = false
                break
            }
        }
        if (isSet) {
            rate[0].valid = true
        }
    }
}

handle(rootConfig.nursePatrolConfig.nurseLevel1Rate)
handle(rootConfig.nursePatrolConfig.nurseLevel2Rate)
handle(rootConfig.nursePatrolConfig.nurseLevel3Rate)

db.mnisReportConfig.deleteOne({})
db.mnisReportConfig.insertOne(rootConfig)
