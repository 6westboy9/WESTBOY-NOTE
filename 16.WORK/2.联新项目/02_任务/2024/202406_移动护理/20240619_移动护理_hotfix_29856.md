    "key": "uniqueOrderBar",

    "componentType": "Input",
    "editable": true,
    "display": true,
    "prefix": "orderbar去重"

    let child = children[i]
    if (child.key !== "ORDER_SEND_DATA_LIST") {
        continue;
    }

    let attrs = child.pattern.optionKeyAttrs;
    if (typeof attrs != 'undefined' && Array.isArray(attrs) && attrs.length > 0) {
        attrs = attrs.filter(function(attr) {
            return attr.key !== "uniqueOrderBar";
        });
        attrs.push(item);
    }
    child.pattern.optionKeyAttrs = attrs;
}
