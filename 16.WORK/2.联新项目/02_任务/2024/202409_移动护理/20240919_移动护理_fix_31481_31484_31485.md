MySQL脚本

```sql
use windranger_mnis;  
  
# 查询并留底含有表体配置数据源的护记单  
CREATE TABLE IF NOT EXISTS doc_template_info_nurse_source_tmp_240919 (select seq_id,template_name,template_code from doc_template_info where template_code in (  
select distinct dti.template_code from doc_template_info dti  
inner join doc_template_item dti2 on dti.template_code = dti2.template_code  
where dti.report_type_code = 'nursingDoc' and dti2.item_type = 'body'  
and (dti2.source_table is not null and dti2.source_table <> '')));  
  
# 清空护记单表体数据源配置  
update doc_template_item  
set source_table  = null,  
    source_column = null,  
    source_update = 0  
where template_code in (select template_code from doc_template_info where report_type_code = 'nursingDoc')  
  and item_type = 'body';
```
