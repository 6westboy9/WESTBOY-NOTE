use coms

var configDoc = db.sysConfigFile.findOne({ 'key': 'orderExecutionConfig' });

if (configDoc && configDoc.children) {
    // 检查是否已存在enableOralDrugScanExecuteToManual
    var existingChild = configDoc.children.find(function(child) {
        return child.key === 'enableOralDrugScanExecuteToManual';
    });

    if (existingChild) {
        print('enableOralDrugScanExecuteToManual 已存在，无需操作');
    } else {
        // 不存在则新增
        var newChild = {
            "_id": "65095ee1cf1fe2309d010848-Mon Dec 15 2025 10:23:23 GMT+0800",
            "key": "enableOralDrugScanExecuteToManual",
            "pattern": {
                "description": "从PDA口服药执行入口扫描患者腕带后，进行医嘱执行时，是否转为手动执行",
                "componentType": "Switch",
                "label": "PDA口服药执行入口是否改为手动执行",
                "urlOptionSourceDetail": {},
                "optionSourceDetail": "",
                "optionSource": "options",
                "options": []
            },
            "value": false
        };

        var result = db.sysConfigFile.update(
            { 'key': 'orderExecutionConfig' },
            { "$push": { "children": newChild } }
        );

        if (result.nModified > 0) {
            print('成功添加 enableOralDrugScanExecuteToManual 配置项');
        } else {
            print('添加失败或文档未找到');
        }
    }
} else {
    print('未找到orderExecutionConfig配置或children为空');
}
