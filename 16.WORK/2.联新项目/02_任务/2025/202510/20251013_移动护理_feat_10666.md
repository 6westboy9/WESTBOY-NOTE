use mnis

// 目标集合
const coll = db.getCollection("patInhosOrderGroupExtInfo");
// 获取已有索引列表
const existingIndexes = coll.getIndexes().map(idx => idx.name);
// 需要创建的索引定义
const indexesToCreate = [
  { key: { orderGroupNo: 1 }, name: "idx_orderGroupNo" },
  { key: { inhosCode: 1 }, name: "idx_inhosCode" }
];

// 逐个检查并创建不存在的索引
indexesToCreate.forEach(idx => {
  if (!existingIndexes.includes(idx.name)) {
    print(`Creating index: ${idx.name}`);
    coll.createIndex(idx.key, { name: idx.name });
  } else {
    print(`Index already exists: ${idx.name}`);
  }
});

print("Index check complete.");
