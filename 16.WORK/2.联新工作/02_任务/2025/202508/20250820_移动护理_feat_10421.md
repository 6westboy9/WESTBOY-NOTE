use coms
(function() {
    function addOption(setting) {
        var caSetting = db.sysConfigFile.findOne({ key: setting, wardCode: '' })
        if (caSetting == null) {
            return
        }
        var children = caSetting.children
        var update = false
        if (Array.isArray(children) && children.length > 0) {
            for (var i = 0; i < children.length; i++) {
                var child = children[i]
                if (child.key === 'caProvider') {
                    var options = child.pattern.options
                    var exists = false
                    var optionsLength = options.length
                    for (var j = 0; j < optionsLength; j++) {
                        var option = options[j]
                        if (option.code === 'gdWztNetCa') {
                            exists = true
                            break
                        }
                    }

                    print(setting, optionsLength, exists)

                    if (!exists) {
                        var item = {
                            "sortedIndex": optionsLength,
                            "code": "gdWztNetCa",
                            "name": "广东网证通CA"
                        }
                        print("不存在进行添加: " + JSON.stringify(item))
                        options.push(item)
                        update = true
                    }
                }
            }
        }

        if (update) {
            db.sysConfigFile.save(caSetting)
        }
    }

    addOption('CASettingSub');
    addOption('CASetting');
})();
