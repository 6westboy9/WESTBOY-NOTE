```js
use mnis

var collectionName = "patInhosOrderGroupExtInfo";
var indexName = "idx_orderGroupNo";
var collection = db.getCollection(collectionName);

// 检查索引是否已存在
var existingIndexes = collection.getIndexes();
var indexExists = false;

for (var i = 0; i < existingIndexes.length; i++) {
    if (existingIndexes[i].name === indexName) {
        indexExists = true;
        break;
    }
}

// 根据检查结果执行相应操作
if (indexExists) {
    print("索引 " + indexName + " 已存在，跳过创建");
} else {
    try {
        collection.createIndex({ "orderGroupNo": 1 }, {
            "name": indexName,
            "background": true  // 使用后台构建，减少对生产环境的影响
        });
        print("索引 " + indexName + " 创建成功");
    } catch (e) {
        print("索引创建失败: " + e.message);
    }
}
```
