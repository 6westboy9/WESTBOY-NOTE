# Redux Toolkit 代码速查表

## 1. Slice 模板

```typescript
import { createSlice, PayloadAction } from '@reduxjs/toolkit';

interface StateType {
  value: number;
}

const initialState: StateType = {
  value: 0,
};

export const sliceName = createSlice({
  name: 'sliceName',
  initialState,
  reducers: {
    // 无参数 action
    simpleAction: (state) => {
      state.value += 1;
    },
    // 带参数 action
    actionWithPayload: (state, action: PayloadAction<number>) => {
      state.value += action.payload;
    },
  },
});

export const { simpleAction, actionWithPayload } = sliceName.actions;
export default sliceName.reducer;
```

## 2. Store 配置模板

```typescript
import { configureStore } from '@reduxjs/toolkit';
import sliceReducer from './features/sliceName';

export const store = configureStore({
  reducer: {
    sliceName: sliceReducer,
  },
});

export type RootState = ReturnType<typeof store.getState>;
export type AppDispatch = typeof store.dispatch;
```

## 3. 类型化 Hooks 模板

```typescript
import { useDispatch, useSelector } from 'react-redux';
import type { RootState, AppDispatch } from './store';

export const useAppDispatch = useDispatch.withTypes<AppDispatch>();
export const useAppSelector = useSelector.withTypes<RootState>();
```

## 4. Provider 包裹

```typescript
import { Provider } from 'react-redux';
import { store } from './store';

<Provider store={store}>
  <App />
</Provider>
```

## 5. 组件中使用

```typescript
// 读取状态
const value = useAppSelector((state) => state.sliceName.value);

// 获取 dispatch
const dispatch = useAppDispatch();

// dispatch action
dispatch(actionCreator(payload));
```

## 6. 常见 Redux Toolkit API

| API | 说明 |
|-----|------|
| `createSlice()` | 创建 slice（状态+reducer+actions） |
| `configureStore()` | 配置 store |
| `createAsyncThunk()` | 创建异步 action |
| `createSelector()` | 创建记忆化选择器 |
| `PayloadAction<T>` | action payload 类型 |
