# 案例

## 1

```js
var n = 100;

function foo() {
	n = 200;
}

foo();
console.log(n); // 200
```

```
200
```

## 2

```js
function foo() {
	console.log(n); // undefined
	var n = 200;
	console.log(n); // 200
}

var n = 100;

foo();
```

```
undefined
200
```
## 3

```js
var n = 100;

function foo1() {
  console.log(n); // 100
}

function foo2() {
  var n = 200;
  console.log(n); // 200
  foo1();
}

foo2();
console.log(n); // 100
```

```
200
100
100
```

## 4

```js
var a = 100;

function foo() {
  console.log(a); // undefined
  return;
  var a = 100;
}

foo();
```

```
undefined
```

## 5

没有写var的时候，JS会将b定义为全局变量。

```js
function foo() {
  b = 100;
}

foo();
console.log(b);
```

```js
100
```

## 6

```js
function foo() {
  var a = b = 100;
}

foo();
console.log(b);
console.log(a);
```

```
100
Uncaught ReferenceError: a is not defined
```
