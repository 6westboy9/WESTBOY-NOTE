```js
// 纯函数的示例

/**
 * 纯函数的特点：
 * 1. 给定相同的输入，总是返回相同的输出
 * 2. 没有副作用（不修改外部状态）
 * 3. 不依赖外部状态
 */

// 示例1：纯函数
function add(a, b) {
    return a + b;
}

// 示例2：纯函数
function multiply(x, y) {
    return x * y;
}

// 示例3：不是纯函数（依赖外部状态）
let count = 0;
function incrementCounter() {
    count += 1;  // 修改外部状态，产生副作用
    return count;
}

// 示例4：不是纯函数（依赖外部状态）
function getRandomNumber() {
    return Math.random();  // 每次调用返回不同的结果
}

// 示例5：纯函数的数组操作
function pureArrayDouble(arr) {
    return arr.map(x => x * 2);  // 返回新数组，不修改原数组
}

// 示例6：非纯函数的数组操作
function impureArrayDouble(arr) {
    arr.forEach((item, index) => {
        arr[index] = item * 2;  // 直接修改原数组，产生副作用
    });
    return arr;
}
```
