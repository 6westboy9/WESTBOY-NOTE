背景

```js
let f1 = function (i) {
    new Promise((resolve, reject) => {
        if (i % 2 === 0) {
            resolve('任务成功得到的数据');
        } else {
            reject('失败的信息');
        }
    }).then(data => {
        console.log(data);
    }).catch(error => {
        console.log(error);
    });
}

f1(0); // 任务成功得到的数据
f1(1); // 失败的信息
```
