use mnis
CREATE TABLE IF NOT EXISTS `pat_inhos_nursing_patrol` (  
  `seq_id` bigint(20) unsigned NOT NULL AUTO_INCREMENT COMMENT '自增长流水号',  
  `ward_code` varchar(20) NOT NULL DEFAULT '' COMMENT '病区代码',  
  `inhos_code` varchar(20) NOT NULL DEFAULT '' COMMENT '住院号',  
  `order_bar` varchar(30) NOT NULL DEFAULT '' COMMENT '医嘱条码',  
  `patrol_status` varchar(32) NOT NULL DEFAULT '' COMMENT '巡视状态',  
  `patrol_time` datetime NOT NULL COMMENT '巡视时间',  
  `patrol_nurse_code` varchar(20) NOT NULL DEFAULT '' COMMENT '巡视护士工号',  
  `dosage` varchar(20) DEFAULT '' COMMENT '用量',  
  `dosage_unit` varchar(20) DEFAULT '' COMMENT '用量单位',  
  `speed_value` varchar(20) DEFAULT '' COMMENT '速度值',  
  `speed_unit` varchar(20) DEFAULT '' COMMENT '速度单位',  
  `duration_hour` int(11) DEFAULT '0' COMMENT '时长-时',  
  `duration_minute` int(11) DEFAULT '0' COMMENT '时长-分',  
  `abnormal_flag` tinyint(4) DEFAULT '0' COMMENT '是否异常 0否1是',  
  `abnormal_msg` varchar(100) DEFAULT '' COMMENT '异常表现',  
  `abnormal_disposal` varchar(100) DEFAULT '' COMMENT '处理措施',  
  `create_time` datetime NOT NULL COMMENT '创建时间',  
  `create_person` varchar(20) NOT NULL DEFAULT '' COMMENT '创建人',  
  `update_time` datetime DEFAULT NULL COMMENT '更新时间',  
  `update_person` varchar(20) DEFAULT '' COMMENT '更新人',  
  `is_deleted` tinyint(4) NOT NULL DEFAULT '0' COMMENT '逻辑删除标记 0-有效 1-删除',  
  PRIMARY KEY (`seq_id`),  
  KEY `idx_pat_inhos_nursing_patrol_bar` (`order_bar`,`patrol_time`),  
  KEY `idx_pat_inhos_nursing_patrol_inhos` (`inhos_code`,`patrol_time`)  
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='护理治疗巡视记录'
