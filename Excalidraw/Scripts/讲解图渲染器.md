const THEME = {
  banner: { fill: "#F3F0FF", stroke: "#6D4AFF", text: "#4320A3" },
  card: { fill: "#EAF4FF", stroke: "#2D79C7", text: "#174B7E" },
  callout: { fill: "#FFF4E5", stroke: "#F08C00", text: "#8A4B00" },
  result: { fill: "#FFE8E8", stroke: "#D94841", text: "#8C2924" },
  table: { fill: "#EAFBF1", stroke: "#2F9E62", text: "#17603A" }
};

const ACTIVE_VIEW = ea.setView("active");
if (!ACTIVE_VIEW || !ACTIVE_VIEW.file) {
  throw new Error("当前没有可用的 Excalidraw 视图。");
}

const drawingPath = ACTIVE_VIEW.file.path;
const specPath = drawingPath.replace(/\.excalidraw\.md$/, ".diagram.json");
const markerPath = drawingPath.replace(/\.excalidraw\.md$/, ".diagram.rendered");

if (app.vault.getAbstractFileByPath(markerPath)) {
  return;
}

const specFile = app.vault.getAbstractFileByPath(specPath);
if (!specFile) {
  throw new Error(`未找到兄弟规格文件：${specPath}`);
}

function ensurePalette(kind, block) {
  const palette = { ...(THEME[kind] || THEME.card) };
  if (block.color) palette.fill = block.color;
  if (block.stroke_color) palette.stroke = block.stroke_color;
  if (block.text_color) palette.text = block.text_color;
  return palette;
}

function setBaseStyle(kind, block) {
  const palette = ensurePalette(kind, block);
  ea.style.strokeColor = palette.stroke;
  ea.style.backgroundColor = palette.fill;
  ea.style.roughness = 0;
  ea.style.strokeWidth = kind === "banner" ? 2 : 1;
  ea.style.roundness = 3;
  ea.style.fontFamily = 4;
  ea.style.fontSize = kind === "banner" ? 24 : 20;
  return palette;
}

function addTextBlock(block, palette) {
  const paddingX = 20;
  const titleY = block.y + 18;
  const bodyY = block.y + 62;
  const titleText = block.subtitle ? `${block.title}\n${block.subtitle}` : block.title;
  ea.style.strokeColor = palette.text;
  ea.style.backgroundColor = "transparent";
  ea.style.fontFamily = 4;
  ea.style.fontSize = block.kind === "banner" ? 24 : 20;
  ea.addText(block.x + paddingX, titleY, titleText || "", {
    width: block.width - paddingX * 2,
    textAlign: block.kind === "banner" ? "center" : "left",
    verticalAlign: "top"
  });
  if (block.body) {
    ea.style.fontSize = 18;
    ea.addText(block.x + paddingX, bodyY, block.body, {
      width: block.width - paddingX * 2,
      textAlign: "left",
      verticalAlign: "top"
    });
  }
}

function makeConnectorLabel(anchorX, anchorY, label, color) {
  if (!label) return;
  ea.style.strokeColor = color || "#7C6F64";
  ea.style.backgroundColor = "transparent";
  ea.style.fontFamily = 4;
  ea.style.fontSize = 16;
  ea.addText(anchorX, anchorY, label, { width: 180, textAlign: "center", verticalAlign: "middle" });
}

async function render(force) {
  const raw = await app.vault.read(specFile);
  const spec = JSON.parse(raw);

  const existing = ea.getViewElements();
  if (existing.length) {
    ea.deleteViewElements(existing);
  }

  ea.reset();
  const blockIds = {};

  for (const block of spec.blocks || []) {
    const palette = setBaseStyle(block.kind, block);
    const shapeId = ea.addRect(block.x, block.y, block.width, block.height);
    blockIds[block.id] = shapeId;
    addTextBlock(block, palette);
  }

  for (const connector of spec.connectors || []) {
    const fromId = blockIds[connector.from];
    const toId = blockIds[connector.to];
    if (!fromId || !toId) continue;

    ea.style.strokeColor = connector.color || "#5F6B7A";
    ea.style.backgroundColor = "transparent";
    ea.style.roughness = 0;
    ea.style.strokeWidth = 2;

    ea.connectObjects(
      fromId,
      connector.from_side || "right",
      toId,
      connector.to_side || "left",
      {
        numberOfPoints: 0,
        startArrowHead: "none",
        endArrowHead: "arrow",
        padding: 24
      }
    );

    if (connector.label) {
      const fromBlock = (spec.blocks || []).find((item) => item.id === connector.from);
      const toBlock = (spec.blocks || []).find((item) => item.id === connector.to);
      if (fromBlock && toBlock) {
        const labelX = (fromBlock.x + toBlock.x) / 2;
        const labelY = Math.min(fromBlock.y, toBlock.y) - 36;
        makeConnectorLabel(labelX, labelY, connector.label, connector.color);
      }
    }
  }

  await ea.addElementsToView();

  const markerBody = JSON.stringify({
    renderedAt: new Date().toISOString(),
    drawingPath,
    specPath,
    force: !!force
  }, null, 2);

  const markerFile = app.vault.getAbstractFileByPath(markerPath);
  if (markerFile) {
    await app.vault.modify(markerFile, markerBody);
  } else {
    await app.vault.create(markerPath, markerBody);
  }

  if (ea.targetView?.forceSave) {
    await ea.targetView.forceSave(true);
  } else if (ea.targetView?.save) {
    await ea.targetView.save();
  }
}

await render(false);
